## A - Engines【计算几何】
这题属于计算几何里面的内容，极角排序。先用atan2算向量与x正轴的夹角，然后对夹角进行排序。之后进行暴力枚举所有的情况
```cpp
#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> pii;
const ll maxn = 1e6+10;
const ll inf = 1e14;
struct node{
	ll x,y;
	 
}r[1100];;
bool cmp(node a,node b){
	return atan2(a.x,a.y) < atan2(b.x,b.y);
}
int main(){
	ll n,x,y;
	cin >> n;
	for(ll i = 0; i < n; i++){
		cin >> r[i].x >> r[i].y;
	}
	ll ans = 0;
	sort(r,r+n,cmp);
	for(ll i = 0; i < n; i++){
		x = r[i].x ; y = r[i].y;
		ans = max(ans,x*x+y*y);
		for(ll j = (i+1)%n; j != i; j = (j+1)%n){
			x += r[j].x ; y += r[j].y;
			ans = max(ans,x*x+y*y);
		}
		
	}
	
	double ans1 = sqrt(ans);
	printf("%.13lf\n",ans1);
	return 0;
}
```

## D - Shortest Path on a Line 【图论】
这是一个考堆优化版的狄杰斯特拉算法的题，只不过加了一些思维在里面，这题需要加入所有i到i-1的权值为0的边，这样从[L,R]中选两个点连成一条边，只需要选边界点就可以表示出L到[L,R]所有点的距离，因为有i到i-1边权为0。还有这题是求1->N的最短距离，其实是一个点从小到大走的过程，所以只需要加上有向边就行。
```cpp
#include <iostream>
#include <queue>
using namespace std;
typedef long long  ll;
typedef pair<ll,ll> pii;
const ll maxn = 1e6+10;
const ll inf = 1e14;
ll N,M;
ll h[maxn],e[maxn],ne[maxn],w[maxn],idx;
ll dis[maxn];

priority_queue<pii,vector<pii>,greater<pii> > heap;

void add(ll a,ll b,ll c){
    e[idx] = b,w[idx] = c,ne[idx] = h[a],h[a] = idx++;
}

ll Dijkstra(){
    heap.push({0,1});
    dis[1] = 0;
    
    while (heap.size()) {
        auto cur = heap.top();heap.pop();
        ll d = cur.first,s = cur.second;
        for(ll i = h[s];i!=-1;i = ne[i]){
            ll j = e[i];
            if(d+w[i] < dis[j]){
                dis[j] = d+w[i];
                heap.push({dis[j],j});
            }
        }
    }
    if(dis[N] == inf ) return -1;
    return dis[N];
}

void init(){
    for(ll i = 1;i<=N;i++) dis[i] = inf;
    memset(h,-1,sizeof(h));
    for(ll i = 1;i<=N;i++){
        add(i,i-1,0);
    }
}

int main(){
    cin>>N>>M;
    init();
    ll a,b,c;
    for(ll i = 0;i<M;i++){
        scanf("%lld%lld%lld",&a,&b,&c);
        add(a,b,c);
    }
    cout<<Dijkstra()<<endl;
    return 0;
}
```
## E - Counting of Trees 【快速幂】
要注意给的数据，第一个必须是0，而且必须只有一个0
```cpp
#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
#include <queue>
#include <map>
#include <cstring>
using namespace std;
typedef long long ll;
const ll maxn = 1e6+10;
const ll inf = 1e16;
const ll mod = 998244353;

ll N;
map<ll,ll> mp;
ll arr[maxn];
ll ksm(ll a,ll b){
    ll res = 1;
    while(b){
        if(b&1) res = res*a%mod;
        a = a*a%mod;
        b>>=1;
    }
    return res;
}


int main(){
    cin>>N;
    ll t;
    for(ll i = 1;i<=N;i++){
        scanf("%lld",&t);
        arr[i] = t;
        mp[t]++;
    }
    ll res;
    if(!(mp[0] == 1) || arr[1] != 0 ) res = 0;
    else{
        res = 1;
        for(auto m:mp){
            if(m.first == 0) continue;
            res = res*ksm(mp[m.first-1], m.second)%mod;
        }
    }
    cout<<(res+mod)%mod<<endl;
    return 0;
}
```

## G - Powerful Discount Tickets 【贪心】
每一张优惠券相当于可以优惠掉某个数x的一半，所以只要每次使用优惠券的时候是对当前最大的数使用就好

```cpp
#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
#include <queue>
#include <map>
#include <cstring>
using namespace std;
typedef long long ll;
const ll maxn = 1e6+10;
const ll inf = 1e16;
const ll mod = 998244353;

int N,M;
int arr[maxn];
priority_queue<int> heap;
int main(){    
    cin>>N>>M;
    for(int i = 0;i<N;i++) scanf("%d",&arr[i]);
    for(int i = 0;i<N;i++) heap.push(arr[i]);
    while(M--){
        int t = heap.top(); heap.pop();
        heap.push(t/2);
    }
    ll res = 0;
    while(heap.size()){
        res += heap.top();heap.pop();
    }
    cout<<res<<endl;
    return 0;
}
```

## I - Lower 【傻逼题】
找出最大下降的连续字段，for循环一遍就好
```cpp

vector<int> res;

    cin>>N;
    for(int i = 0;i<N;i++){
        scanf("%d",&arr[i]);
    }
    reverse(arr, arr+N);
    int res = 0,cur = 0;
    for(int i = 1;i<N;i++){
        if(arr[i]>=arr[i-1]){
            cur++;
        }else{
            cur = 0;
        }
        res = max(res,cur);
    }
```

```cpp
    ll res = 1;
    while(b){
        if(b&1) res = res*a%mod;
        a = a*a%mod;
        b>>=1;
    }
    return res;
}

int main(){
    cin>>N>>K;

        scanf("%lld",&arr[i]);
    }
    for(int i = 0;i<N;i++){
        for(int j = 0;j<N;j++){
            if(arr[i]>arr[j]) bigR[i]++;
        }

            if(arr[j]<arr[i]) bigL[i]++;
        }
    }

    for(int i = 0;i<N;i++){
        res = (res + K*(bigR[i]+K*bigR[i]%mod)%mod*ksm(2,mod-2)%mod)%mod;
        res = (res - K*bigL[i] + mod)%mod;
    }
    cout<<(res+mod)%mod<<endl;
    
    
    return 0;
}
```

## M - AB Substrings 【思维】

```cpp
#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
#include <queue>

typedef long long ll;
const ll maxn = 1e6+10;
const ll inf = 1e16;
const ll mod = 998244353;

int N;
int cntA,cntB,rd;
int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);cout.tie(0);
    cin>>N;
    string s;
    ll res = 0;
    while(N--){
        cin>>s;
        if(s[0] == 'B' && s[s.length()-1] == 'A') rd++;
        if(s[0] == 'B') cntB++;
        if(s[s.length()-1] == 'A') cntA++;
        for(int i = 1;i<s.length();i++){
            if(s[i] == 'B' && s[i-1] == 'A') res++;
        }
    }
    res = res + min(cntA,cntB);
    if(cntA == cntB && cntA == rd && cntA) res--;
    cout<<res<<endl;

    return 0;
}
```
